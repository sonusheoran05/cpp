5) C program to enter student marks and find percentage and grade
A college has the following rules for the grading system:
5 marks (grade + %)
1. Below 25 – F
2. 25 to 45 – E
3. 45 to 50 – D
4. 50 to 60 – C
5. 60 to 80 – B 
6. Above 80 – A

#include<iostream>
using namespace std;
int main()
{
	int m;
	cout<<"enter marks m";
	cin>>m;
	int x;
	cout<<"enter marks x";
	cin>>x;
	int y;
	cout<<"enter marks y";
	cin>>y;
	{
		int t,a;
		t=m+x+y;
		a=(t*100)/300;
		cout<<"enter percentage"<<a<<endl;
		{
			if(a<25)
			{
				cout<<"F";
			}
			if(a>=25&&a<=45)
			{
				cout<<"E";
			}
			if(a>=45&&a<=50)
			{
				cout<<"D";
			}
			if(a>=50&&a<=60)
			{
				cout<<"C";
			}
			if(a>=60&&a<=80)
			{
				cout<<"B";
			}
			if(a>=80)
			{
				cout<<"A";
			}
		}
	}
		return 0;
}
